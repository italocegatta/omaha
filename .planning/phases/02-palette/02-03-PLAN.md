def test_all_tokens_pass_aa() -> None:

    if not APP_CSS_PATH.exists():

def test_corrected_pairs_pass_aa() -> None:

    pairs: list[tuple[str, str, str, float]] = [
        # (label, foreground, background, minimum_ratio)
        ("class-4 on bg", "oklch(0.53 0.13 50)", "oklch(0.975 0.003 60)", 4.5),
        ("class-6 on bg", "oklch(0.52 0.10 200)", "oklch(0.975 0.003 60)", 4.5),
        ("negative-ink on negative", "#ffffff", "oklch(0.50 0.18 25)", 4.5),
        ("error-fg on error-bg", "oklch(0.38 0.12 25)", "oklch(0.94 0.02 15)", 4.5),
        ("accent-ink on accent", "oklch(0.98 0.005 150)", "oklch(0.42 0.09 150)", 4.5),
        ("ink on bg", "oklch(0.20 0.01 60)", "oklch(0.975 0.003 60)", 4.5),
    ]
    for label, fg, bg, threshold in pairs:
        ratio = contrast_ratio(fg, bg)
        _, status = aa_status(ratio)
        assert ratio >= threshold, (
            f"{label}: ratio={ratio:.2f} < threshold={threshold}"
        )
        assert status == "Passa", (
            f"{label}: expected Passa, got {status} (ratio={ratio:.2f})"
        )
```

def test_negative_ink_present() -> None:
    """--negative-ink token is defined in app.css :root block."""
    if not APP_CSS_PATH.exists():
        pytest.skip("app.css not found")
    sheet = parse_stylesheet(APP_CSS_PATH)
    # Find the :root rule
    root_rule = None
    for rule in sheet.rules:
        if rule.selector == ":root":
            root_rule = rule
            break
    assert root_rule is not None, ":root rule not found in app.css"
    assert "--negative-ink" in root_rule.declarations, (
        f"--negative-ink not in :root. Found: {list(root_rule.declarations.keys())}"
    )
    assert "--positive-ink" in root_rule.declarations, (
        "--positive-ink not in :root"
    )
```

def test_legacy_aliases_intact() -> None:
    """--fg and --muted legacy aliases still resolve correctly."""
    if not APP_CSS_PATH.exists():
        pytest.skip("app.css not found")
    sheet = parse_stylesheet(APP_CSS_PATH)
    root_rule = None
    for rule in sheet.rules:
        if rule.selector == ":root":
            root_rule = rule
            break
    assert root_rule is not None, ":root rule not found"
    fg_val = root_rule.declarations.get("--fg", "")
    muted_val = root_rule.declarations.get("--muted", "")
    # Aliases must still point to their targets
    assert "var(--ink)" in fg_val, f"--fg does not alias --ink: got {fg_val}"
    assert "var(--ink-muted)" in muted_val, f"--muted does not alias --ink-muted: got {muted_val}"
```

def test_hardcoded_white_removed() -> None:
    """No hardcoded color: #fff in component delete-confirm rules."""
    if not APP_CSS_PATH.exists():
        pytest.skip("app.css not found")
    content = APP_CSS_PATH.read_text()
    lines = content.split("\n")
    # Find lines inside .class-delete-confirm-yes and .dashboard-asset-delete-confirm-yes rules
    # Check that color: var(--negative-ink) is present and color: #fff is absent
    assert "color: var(--negative-ink)" in content, (
        "var(--negative-ink) not found in app.css"
    )
    # Check for hardcoded #fff in the specific context of confirm-yes rules
    in_delete_confirm = False
    hardcoded_found = []
    for i, line in enumerate(lines, 1):
        stripped = line.strip()
        if "delete-confirm-yes" in stripped and "{" in stripped:
            in_delete_confirm = True
        elif in_delete_confirm and "}" in stripped:
            in_delete_confirm = False
        elif in_delete_confirm and ("color: #fff" in stripped.lower() or "color:#fff" in stripped.lower()):
            hardcoded_found.append(f"Line {i}: {stripped}")
    assert len(hardcoded_found) == 0, (
        f"Hardcoded #fff found in delete-confirm rules: {hardcoded_found}"
    )
```
